## 📝 صياغة سريعة (Cheat Sheet)

```sifr
// متغيّرات
كنز الاسم = "محمد"
سرّ السنة = 1446

// دالة
مهمّة جمع(أ، ب) :
  أعد أ + ب
انتهى

// شرط
إن السنة > 1440 :
  أرني("جديدة")
وإلا :
  أرني("قديمة")
انتهى

// حلقة
جوال ع من 1 إلى 10 :
  أرني(ع)
انتهى

// قائمة وتكرار
كنز أسماء = ["محمد", "علي", "فاطمة"]
جوال اسم في أسماء :
  أرني("مرحباً " + اسم)
انتهى

// بنية
بنية شخص :
  اسم: نص
  عمر: رقم
  مهمّة عرّف(هذا) :
    أرني(هذا.اسم + " عمره " + هذا.عمر)
  انتهى
انتهى

// معالجة الأخطاء
حاول :
  كنز ن = 10 / 0
التقط خ :
  أرني("خطأ: " + خ)
انتهى
```
